16  package org.onehippo.forge.folderctxmenus.cms.plugin;
17  
18  import com.fasterxml.jackson.databind.JsonNode;
19  import com.fasterxml.jackson.databind.ObjectMapper;
20  import org.junit.Before;
21  import org.junit.Test;
22  
23  import static org.junit.Assert.*;
24  
25  public class ProgressPanelJsonTest {
26  
27      private static final ObjectMapper MAPPER = new ObjectMapper();
28      private ProgressTrackingOperationProgress progress;
29  
30      @Before
31      public void setUp() {
32          progress = new ProgressTrackingOperationProgress();
33      }
34  
35      @Test
36      public void buildProgressPayload_shouldContainAllFields() throws Exception {
37          progress.updateProgress(50, 100, "/content/documents/test");
38  
39          String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
40          JsonNode node = MAPPER.readTree(json);
41  
42          assertEquals(50, node.get("current").asLong());
43          assertEquals(100, node.get("total").asLong());
44          assertEquals(50, node.get("percent").asInt());
45          assertEquals("/content/documents/test", node.get("path").asText());
46          assertTrue(node.has("eta"));
47          assertFalse(node.get("cancelled").asBoolean());
48          assertFalse(node.get("completed").asBoolean());
49      }
50  
51      @Test
52      public void buildProgressPayload_withCompletionSummary_shouldIncludeSummaryObject() throws Exception {
53          progress.updateProgress(100, 100, "/done");
54          progress.setCompletionSummary(new ProgressCompletionSummary("Successfully copied 100 items", false));
55          progress.markCompleted();
56  
57          String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
58          JsonNode node = MAPPER.readTree(json);
59  
60          assertTrue(node.has("summary"));
61          JsonNode summary = node.get("summary");
62          assertEquals("Successfully copied 100 items", summary.get("message").asText());
63          assertFalse(summary.get("error").asBoolean());
64      }
65  
66      @Test
67      public void buildProgressPayload_withErrorSummary_shouldIncludeErrorFlag() throws Exception {
68          progress.updateProgress(50, 100, "/error");
69          progress.setCompletionSummary(new ProgressCompletionSummary("Error after 50 items", true));
70  
71          String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
72          JsonNode node = MAPPER.readTree(json);
73  
74          assertTrue(node.get("summary").get("error").asBoolean());
75      }
76  
77      @Test
78      public void buildProgressPayload_shouldEscapeSpecialCharacters() throws Exception {
79          progress.updateProgress(1, 10, "/path/with\"quotes\\and\nspecial");
80  
81          String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
82          JsonNode node = MAPPER.readTree(json);
83  
84          assertEquals("/path/with\"quotes\\and\nspecial", node.get("path").asText());
85      }
86  
87      @Test
88      public void buildProgressPayload_withNullPath_shouldHandleGracefully() throws Exception {
89          progress.updateProgress(0, 0, null);
90  
91          String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
92          JsonNode node = MAPPER.readTree(json);
93  
94          assertTrue(node.get("path").isNull());
95      }
96  
97      @Test
98      public void buildProgressPayload_whenCancelled_shouldReflectCancelledState() throws Exception {
99          progress.updateProgress(25, 100, "/test");
100         progress.cancel();
101 
102         String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
103         JsonNode node = MAPPER.readTree(json);
104 
105         assertTrue(node.get("cancelled").asBoolean());
106     }
107 
108     @Test
109     public void buildProgressPayload_duringFinalizingPhase_shouldEmitFinalizingFields() throws Exception {
110         progress.updateProgress(100, 100, "/done");
111         progress.enterFinalizingPhase();
112         progress.updateFinalizingProgress(42);
113 
114         String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
115         JsonNode node = MAPPER.readTree(json);
116 
117         assertTrue(node.get("finalizing").asBoolean());
118         assertEquals(42, node.get("finalizingCount").asLong());
119         assertFalse(node.has("current"));
120         assertFalse(node.has("total"));
121         assertFalse(node.has("path"));
122     }
123 
124     @Test
125     public void buildProgressPayload_duringFinalizingPhase_withCompletionSummary_shouldIncludeBothFinalizingAndSummary() throws Exception {
126         progress.enterFinalizingPhase();
127         progress.updateFinalizingProgress(42);
128         progress.setCompletionSummary(new ProgressCompletionSummary("Successfully copied 42 items", false));
129         progress.markCompleted();
130 
131         String json = ProgressPanelJsonBuilder.buildProgressPayload(progress);
132         JsonNode node = MAPPER.readTree(json);
133 
134         // finalizing fields must still be present
135         assertTrue(node.get("finalizing").asBoolean());
136         assertEquals(42, node.get("finalizingCount").asLong());
137         // summary must ALSO be present alongside finalizing so the JS
138         // summary-first check can stop polling after completion
139         assertTrue("summary must be present even while finalizing", node.has("summary"));
140         assertEquals("Successfully copied 42 items", node.get("summary").get("message").asText());
141         assertFalse(node.get("summary").get("error").asBoolean());
142         assertTrue(node.get("completed").asBoolean());
143     }
144 
145     @Test
146     public void buildPollingConfig_shouldContainAllElementIds() throws Exception {
147         ProgressPanelJsonBuilder.PollingConfig config = new ProgressPanelJsonBuilder.PollingConfig(
148                 "http://callback",
149                 "panel1",
150                 200,
151                 60,
152                 "status1",
153                 "bar1",
154                 "label1",
155                 "path1",
156                 "cancel1",
157                 "close1"
158         );
159 
160         String json = ProgressPanelJsonBuilder.buildPollingConfig(config);
161         JsonNode node = MAPPER.readTree(json);
162 
163         assertEquals("http://callback", node.get("url").asText());
164         assertEquals("panel1", node.get("panelId").asText());
165         assertEquals(200, node.get("intervalMs").asInt());
166         assertEquals(60, node.get("maxPathLength").asInt());
167         assertEquals("status1", node.get("statusId").asText());
168         assertEquals("bar1", node.get("progressBarId").asText());
169         assertEquals("label1", node.get("progressLabelId").asText());
170         assertEquals("path1", node.get("pathId").asText());
171         assertEquals("cancel1", node.get("cancelId").asText());
172         assertEquals("close1", node.get("closeId").asText());
173     }
174 }
