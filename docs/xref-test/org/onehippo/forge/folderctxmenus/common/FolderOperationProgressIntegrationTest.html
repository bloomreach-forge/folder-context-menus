33  public class FolderOperationProgressIntegrationTest {

35      private BrxmTestingRepository repository;
36      private Session session;
37      private Node sourceFolderNode;
38      private Node destParentFolderNode;
39      private TestOperationProgress progress;
40  
41      @BeforeEach
42      public void setUp() throws Exception {
43          repository = new BrxmTestingRepository();
44          session = repository.login(new SimpleCredentials("admin", "admin".toCharArray()));
45  
46          Node root = session.getRootNode();
47  
48          if (!root.hasNode("content")) {
49              root.addNode("content", NodeType.NT_FOLDER);
50          }
51          Node contentNode = root.getNode("content");
52  
53          if (!contentNode.hasNode("documents")) {
54              contentNode.addNode("documents", NodeType.NT_FOLDER);
55          }
56          Node documentsNode = contentNode.getNode("documents");
57  
58          if (documentsNode.hasNode("testfolder")) {
59              documentsNode.getNode("testfolder").remove();
60          }
61          sourceFolderNode = documentsNode.addNode("testfolder", NodeType.NT_FOLDER);
62  
63          sourceFolderNode.addNode("child1", NodeType.NT_FOLDER);
64          sourceFolderNode.addNode("child2", NodeType.NT_FOLDER);
65          sourceFolderNode.getNode("child1").addNode("grandchild", NodeType.NT_FOLDER);
66  
67          if (documentsNode.hasNode("archive")) {
68              documentsNode.getNode("archive").remove();
69          }
70          destParentFolderNode = documentsNode.addNode("archive", NodeType.NT_FOLDER);
71  
72          session.save();
73  
74          progress = new TestOperationProgress();
75      }
76  
77      @AfterEach
78      public void tearDown() throws Exception {
79          if (session != null) {
80              session.logout();
81          }
82          if (repository != null) {
83              repository.close();
84          }
85      }

87      @Test
88      public void testCopyFolderWithProgress() throws RepositoryException {
89          FolderCopyTask task = new FolderCopyTask(
90                  session,
91                  Locale.ENGLISH,
92                  sourceFolderNode,
93                  destParentFolderNode,
94                  "testfolder-copy",
95                  "Test Folder Copy",
96                  false
97          );
98          task.setOperationProgress(progress);
99  
100         task.execute();
101 
102         assertTrue(progress.updateCount > 0, "Progress should have been updated");
103         assertTrue(progress.hookCallCount > 0, "onProgressUpdated hook should have been called");
104         assertEquals(progress.updateCount, progress.hookCallCount, "Hook should be called once per update");
105         assertTrue(progress.lastTotal > 0, "Total should be greater than zero");
106         assertFalse(progress.cancelled, "Should not be cancelled");
107 
108         assertTrue(destParentFolderNode.hasNode("testfolder-copy"), "Copied folder should exist");
109         Node copiedFolder = destParentFolderNode.getNode("testfolder-copy");
110         assertTrue(copiedFolder.hasNode("child1"), "Copied folder should have children");
111     }
112 
113     @Test
114     public void testMoveFolderWithProgress() throws RepositoryException {
115         FolderMoveTask task = new FolderMoveTask(
116                 session,
117                 Locale.ENGLISH,
118                 sourceFolderNode,
119                 destParentFolderNode,
120                 "testfolder-moved",
121                 "Test Folder Moved"
122         );
123         task.setOperationProgress(progress);
124 
125         String originalPath = sourceFolderNode.getPath();
126         task.execute();
127 
128         assertTrue(progress.updateCount > 0, "Progress should have been updated");
129         assertTrue(progress.hookCallCount > 0, "onProgressUpdated hook should have been called");
130         assertTrue(progress.lastTotal > 0, "Total should be greater than zero");
131         assertFalse(progress.cancelled, "Should not be cancelled");
132 
133         assertFalse(session.nodeExists(originalPath), "Original folder should not exist");
134         assertTrue(destParentFolderNode.hasNode("testfolder-moved"), "Moved folder should exist");
135         Node movedFolder = destParentFolderNode.getNode("testfolder-moved");
136         assertTrue(movedFolder.hasNode("child1"), "Moved folder should have children");
137     }

139     @Test
140     public void testCopyFolderWithCancellation() throws RepositoryException {
141         CancellableOperationProgress cancellableProgress = new CancellableOperationProgress(2);
142 
143         FolderCopyTask task = new FolderCopyTask(
144                 session,
145                 Locale.ENGLISH,
146                 sourceFolderNode,
147                 destParentFolderNode,
148                 "testfolder-cancelled",
149                 "Test Folder Cancelled",
150                 false
151         );
152         task.setOperationProgress(cancellableProgress);
153 
154         assertThrows(OperationCancelledException.class, task::execute, "Should have thrown OperationCancelledException");
155         assertTrue(cancellableProgress.cancelled, "Should have been cancelled");
156         assertTrue(cancellableProgress.updateCount > 0, "Should have recorded some progress");
157     }

183     public void testProgressReportsAccurateCount() throws RepositoryException {
184         FolderCopyTask task = new FolderCopyTask(
185                 session,

202     private static class TestOperationProgress implements OperationProgress {
203         int updateCount = 0;
204         int hookCallCount = 0;
205         long lastCurrent = 0;
206         long lastTotal = 0;
207         String lastPath = null;
208         boolean cancelled = false;
209         boolean finalizingEntered = false;
210         long lastFinalizingCount = 0;
211 
212         @Override
213         public void updateProgress(long current, long total, String currentItemPath) {
214             updateCount++;
215             lastCurrent = current;
216             lastTotal = total;
217             lastPath = currentItemPath;
218         }
219 
220         @Override
221         public boolean isCancelled() {
222             return cancelled;
223         }
224 
225         @Override
226         public void onProgressUpdated() {
227             hookCallCount++;
228         }
229 
230         @Override
231         public void enterFinalizingPhase() {
232             finalizingEntered = true;
233         }
234 
235         @Override
236         public void updateFinalizingProgress(long count) {
237             lastFinalizingCount = count;
238         }
239     }
240 
241     private static class CancellableOperationProgress extends TestOperationProgress {
242         private final int cancelAfterUpdates;
243 
244         CancellableOperationProgress(int cancelAfterUpdates) {
245             this.cancelAfterUpdates = cancelAfterUpdates;
246         }
247 
248         @Override
249         public void updateProgress(long current, long total, String currentItemPath) {
250             super.updateProgress(current, total, currentItemPath);
251             if (updateCount >= cancelAfterUpdates) {
252                 cancelled = true;
253             }
254         }
255     }
