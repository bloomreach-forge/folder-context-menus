113     public CopyOrMoveFolderDialog(final IPluginContext pluginContext, final IPluginConfig pluginConfig,
114                                   IModel<String> titleModel, IModel<FolderActionDocumentArguments> model,
115                                   boolean isCopyDialog) {
116 
117         super(pluginContext, pluginConfig, titleModel, model);
118 
119         this.isCopyOperation = isCopyDialog;
120         final FolderActionDocumentArguments folderActionDocumentModel = model.getObject();
121 
122         boolean isSourceFolderTranslated = false;
123         if (StringUtils.isNotEmpty(folderActionDocumentModel.getSourceFolderIdentifier())) {
124             try {
125                 Node sourceFolderNode = UserSession.get().getJcrSession().getNodeByIdentifier(folderActionDocumentModel.getSourceFolderIdentifier());
126                 sourceFolderIdentifier = sourceFolderNode.getIdentifier();
127                 sourceFolderPath = sourceFolderNode.getPath();
128                 sourceFolderPathDisplay = getDisplayPathOfNode(sourceFolderNode);
129                 newFolderName = getDisplayNameOfNode(sourceFolderNode);
130                 newFolderUrlName = folderActionDocumentModel.getSourceFolderUriName();
131                 isSourceFolderTranslated = sourceFolderNode.isNodeType("hippotranslation:translated");
132             } catch (RepositoryException e) {
133                 log.error("Failed to retrieve source folder node.", e);
134             }
135         }
136 
137         formContainer = new WebMarkupContainer("formContainer");
138         formContainer.setOutputMarkupId(true);
139         formContainer.setOutputMarkupPlaceholderTag(true);
140         add(formContainer);
141 
142         progressContainer = new WebMarkupContainer("progressContainer");
143         progressContainer.setOutputMarkupId(true);
144         progressContainer.setOutputMarkupPlaceholderTag(true);
145         progressContainer.setVisible(false);
146         progressPanel = new ProgressPanel("progressPanel", new ProgressTrackingOperationProgress(), null) {
147             private static final long serialVersionUID = 1L;
148 
149             @Override
150             protected void onCloseClicked(AjaxRequestTarget target) {
151                 refreshJcrSession();
152                 closeDialog();
153                 browseToDestinationIfSuccessful();
154             }
155         };
156         progressPanel.setVisible(false);
157         progressContainer.add(progressPanel);
158         add(progressContainer);
159 
160         final Form form = new Form("form");
161         formContainer.add(form);
162 
163         final TextField<String> sourceFolderPathDisplayField =
164             new TextField<>("sourceFolderPathDisplay", LambdaModel.of(this::getSourceFolderPathDisplay, this::setSourceFolderPathDisplay));
165         sourceFolderPathDisplayField.setEnabled(false);
166         form.add(sourceFolderPathDisplayField);
167 
168         final TextField<String> destinationFolderPathDisplayField =
169             new TextField<>("destinationFolderPathDisplay", LambdaModel.of(this::getDestinationFolderPathDisplay, this::setDestinationFolderPathDisplay));
170         destinationFolderPathDisplayField.setEnabled(false);
171         destinationFolderPathDisplayField.setOutputMarkupId(true);
172         form.add(destinationFolderPathDisplayField);
173 
174         rootNodeModel = new JcrNodeModel(sourceFolderPath.replaceAll("^(/[^/]+/[^/]+).*", "$1"));
175         FolderOnlyDocumentListFilter folderOnlyTreeConfig = new FolderOnlyDocumentListFilter(pluginConfig);
176         rootTreeNode = new FolderTreeNode(rootNodeModel, folderOnlyTreeConfig);
177         treeModel = new JcrTreeModel(rootTreeNode);
178         folderTree = new FolderSelectionCmsJcrTree("folderTree", treeModel, pluginContext, pluginConfig);
179         folderTree.setRootLess(true);
180         folderTree.addTreeNodeEventListener(new ITreeNodeEventListener() {
181             private static final long serialVersionUID = 1L;
182 
183             @Override
184             public void nodeLinkClicked(AjaxRequestTarget target, TreeNode clickedNode) {
185                 if (clickedNode instanceof IJcrTreeNode) {
186                     try {
187                         final IJcrTreeNode treeNodeModel = (IJcrTreeNode) clickedNode;
188                         final Node folderNode = treeNodeModel.getNodeModel().getObject();
189                         destinationFolderIdentifier = folderNode.getIdentifier();
190                         destinationFolderPath = folderNode.getPath();
191                         destinationFolderPathDisplay = getDisplayPathOfNode(folderNode);
192                         target.add(destinationFolderPathDisplayField);
193                     } catch (RepositoryException e) {
194                         log.error("Failed to retrieve selected folder node.", e);
195                     }
196                 }
197             }
198         });
199 
200         form.add(folderTree);
201 
202         final TextField<String> newFolderUrlNameField =
203             new TextField<>("newFolderUrlName", LambdaModel.of(this::getNewFolderUrlName, this::setNewFolderUrlName));
204         newFolderUrlNameField.setOutputMarkupId(true);
205         form.add(newFolderUrlNameField);
206 
207         final TextField<String> newFolderNameField =
208             new TextField<>("newFolderName", LambdaModel.of(this::getNewFolderName, this::setNewFolderName));
209         newFolderNameField.setOutputMarkupId(true);
210         newFolderNameField.add(new AjaxFormComponentUpdatingBehavior("change") {
211             private static final long serialVersionUID = 1L;
212 
213             @Override
214             protected void onUpdate(final AjaxRequestTarget target) {
215                 newFolderUrlName = new StringCodecFactory.UriEncoding().encode(getNewFolderName());
216                 target.add(newFolderUrlNameField);
217             }
218         });
219         form.add(newFolderNameField);
220 
221         WebMarkupContainer row = new WebMarkupContainer("linkAsTranslationRow");
222         row.setVisible(isCopyDialog && isSourceFolderTranslated); // this will remove the <tr> from the HTML output
223         form.add(row);
224         linkAsTranslation = pluginConfig.getAsBoolean(DEFAULT_LINK_TRANSLATION);
225         final CheckBox linkAsTranslationsField =
226             new CheckBox("linkAsTranslation", LambdaModel.of(this::getLinkAsTranslation, this::setLinkAsTranslation));
227         linkAsTranslationsField.setOutputMarkupId(true);
228         linkAsTranslationsField.setVisible(isCopyDialog && isSourceFolderTranslated);
229         row.add(linkAsTranslationsField);
230     }

295     protected void startOperationWithProgress(AjaxRequestTarget target, FolderOperationExecutor executor) {
296         inProgressMode = true;
297         operationProgress = new ProgressTrackingOperationProgress();
298         operationError.set(null);
299         outcomeLogged.set(false);
300         operationUserId = resolveUserId();
301         operationId = UUID.randomUUID().toString();
302 
303         formContainer.setVisible(false);
304         setOkVisible(false);
305         setCancelVisible(false);
306 
307         progressContainer.setVisible(true);
308         progressPanel.reinitialize(operationProgress, () -> {
309             if (operationProgress.isCancelled()) {
310                 operationProgress.setCompletionSummary(buildCompletionSummary(operationError.get()));
311                 operationProgress.markCompleted();
312                 return;
313             }
314             CompletableFuture.runAsync(() -> {
315                 try {
316                     executor.execute(operationProgress);
317                 } catch (Exception e) {
318                     operationError.set(e);
319                 } finally {
320                     Exception error = operationError.get();
321                     logOutcome(error);
322                     operationProgress.setCompletionSummary(buildCompletionSummary(error));
323                     operationProgress.markCompleted();
324                 }
325             }, FOLDER_OP_EXECUTOR);
326         });
327         progressPanel.setVisible(true);
328 
329         target.add(formContainer, progressContainer);
330         target.appendJavaScript("FolderContextMenus.resizeDialog(" + PROGRESS_DIALOG_WIDTH + ")");
331 
332     }

334     protected boolean validateInput() {
335         if (StringUtils.isBlank(destinationFolderIdentifier)) {
336             error("Please select the target folder.");
337             return false;
338         }
339         if (StringUtils.isBlank(newFolderUrlName) || StringUtils.isBlank(newFolderName)) {
340             error("Please enter the destination folder name.");
341             return false;
342         }
343         return true;
344     }
345 
346     protected boolean destinationExists(Session session) {
347         try {
348             Node destParentNode = session.getNodeByIdentifier(destinationFolderIdentifier);
349             String destPath = destParentNode.getPath() + "/" + newFolderUrlName;
350             if (session.nodeExists(destPath)) {
351                 error("A folder named '" + newFolderName + "' already exists at the destination. " +
352                         "Please choose a different name or destination.");
353                 return true;
354             }
355         } catch (RepositoryException e) {
356             log.error("Failed to check destination folder", e);
357             error("Failed to verify destination: " + e.getMessage());
358             return true;
359         }
360         return false;
361     }

363     private static String getUserFriendlyErrorMessage(Exception e) {
364         Throwable cause = e;
365         while (cause != null) {
366             if (cause instanceof AccessDeniedException) {
367                 return "Access denied. You may not have permission to perform this operation.";
368             }
369             if (cause instanceof LockException) {
370                 return "The content is currently locked by another user or process.";
371             }
372             if (cause instanceof ConstraintViolationException) {
373                 return "The operation violates repository constraints.";
374             }
375             cause = cause.getCause();
376         }
377 
378         if (e instanceof RepositoryException) {
379             return "A repository error occurred. Please try again or contact support.";
380         }
381         return "An unexpected error occurred.";
382     }

384     private ProgressCompletionSummary buildCompletionSummary(Exception error) {
385         long count = operationProgress.getCurrentCount();
386         boolean cancelled = error instanceof OperationCancelledException || operationProgress.isCancelled();
387         String verb = isCopyOperation ? "copied" : "moved";
388 
389         if (cancelled) {
390             return new ProgressCompletionSummary("Cancelled after " + verb + " " + count + " items", false);
391         }
392         if (error != null) {
393             return new ProgressCompletionSummary(
394                     "Error after " + count + " items: " + getUserFriendlyErrorMessage(error),
395                     true);
396         }
397         return new ProgressCompletionSummary("Successfully " + verb + " " + count + " items", false);
398     }
399 
400     private void logOutcome(Exception error) {
401         if (!outcomeLogged.compareAndSet(false, true)) {
402             return;
403         }
404 
405         long count = operationProgress.getCurrentCount();
406         boolean cancelled = error instanceof OperationCancelledException || operationProgress.isCancelled();
407 
408         if (cancelled) {
409             log.info("Folder {} operation was cancelled by user", isCopyOperation ? "copy" : "move");
410         } else if (error != null) {
411             log.error("Folder {} operation failed", isCopyOperation ? "copy" : "move", error);
412         } else {
413             logOperationEvent(count);
414         }
415     }
416 
417     private void logOperationEvent(long itemCount) {
418         try {
419             String destJcrPath = destinationFolderPath + "/" + newFolderUrlName;
420             String operation = isCopyOperation ? "copy" : "move";
421 
422             log.info("Folder {} completed: opId='{}', user='{}', source='{}', destination='{}', items={}",
423                     operation,
424                     operationId,
425                     operationUserId,
426                     sourceFolderPath,
427                     destJcrPath,
428                     itemCount);
429         } catch (Exception e) {
430             log.warn("Failed to log folder operation event: {}", e.getMessage(), e);
431         }
432     }

434     @SuppressWarnings("unchecked")
435     private void browseToDestinationIfSuccessful() {
436         if (operationProgress == null || operationProgress.isCancelled() || operationError.get() != null) {
437             return;
438         }
439         if (StringUtils.isBlank(destinationFolderPath) || StringUtils.isBlank(newFolderUrlName)) {
440             return;
441         }
442         IBrowseService<JcrNodeModel> browseService = getPluginContext()
443                 .getService(IBrowseService.class.getName(), IBrowseService.class);
444         if (browseService != null) {
445             browseService.browse(new JcrNodeModel(destinationFolderPath + "/" + newFolderUrlName));
446         }
447     }

449     private void refreshJcrSession() {
450         try {
451             UserSession.get().getJcrSession().refresh(false);
452         } catch (RepositoryException e) {
453             log.warn("Failed to refresh JCR session after folder operation", e);
454         }
455     }
456 
457     private String resolveUserId() {
458         try {
459             return UserSession.get().getJcrSession().getUserID();
460         } catch (Exception e) {
461             log.warn("Failed to resolve user id for folder operation logging", e);
462             return "unknown";
463         }
