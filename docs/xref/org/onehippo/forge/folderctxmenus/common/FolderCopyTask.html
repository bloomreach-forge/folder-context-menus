21  import java.util.UUID;

23  import javax.jcr.ItemNotFoundException;

25  import javax.jcr.RepositoryException;
26  import javax.jcr.Session;
27  import javax.jcr.Value;

29  import org.apache.commons.lang3.ArrayUtils;

31  import org.hippoecm.repository.util.JcrUtils;

33  public class FolderCopyTask extends AbstractFolderCopyOrMoveTask {
34  
35      public FolderCopyTask(final Session session, final Locale locale, final Node sourceFolderNode,
36              final Node destParentFolderNode, final String destFolderNodeName, final String destFolderDisplayName) {
37          super(session, locale, sourceFolderNode, destParentFolderNode, destFolderNodeName, destFolderDisplayName);
38      }
39  
40      @Override
41      protected void doExecute() throws RepositoryException {
42          if (getSourceFolderNode().getParent().isSame(getDestParentFolderNode())) {
43              if (StringUtils.equals(getSourceFolderNode().getName(), getDestFolderNodeName())) {
44                  throw new RuntimeException("Cannot copy to the same folder: " + getDestParentFolderNode().getPath()
45                          + " / " + getDestFolderNodeName());
46              }
47          }
48  
49          if (getSourceFolderNode().isSame(getDestParentFolderNode())) {
50              throw new RuntimeException("Cannot copy to the folder itself: " + getDestFolderPath());
51          }
52  
53          if (getSession().nodeExists(getDestFolderPath())) {
54              throw new RuntimeException("Destination folder already exists: " + getDestFolderPath());
55          }
56  
57          getLogger().info("Copying nodes: from {} to {}.", getSourceFolderNode().getPath(), getDestFolderPath());
58  
59          if (getCopyHandler() != null) {
60              JcrCopyUtils.copy(getSourceFolderNode(), getDestFolderNodeName(), getDestParentFolderNode(),
61                      getCopyHandler());
62          } else {
63              JcrCopyUtils.copy(getSourceFolderNode(), getDestFolderNodeName(), getDestParentFolderNode());
64          }
65  
66          setDestFolderNode(JcrUtils.getNodeIfExists(getDestParentFolderNode(), getDestFolderNodeName()));
67  
68          updateFolderTranslations(getDestFolderNode(), getDestFolderDisplayName(), getLocale().getLanguage());
69      }
70  
71      @Override
72      protected void doAfterExecute() throws RepositoryException {
73          resetHippoDocBaseLinks();
74          takeOfflineHippoDocs();
75          resetHippoDocumentTranslationIds();
76      }
77  
78      /**
79       * Search all the hippotranslation:translated nodes under {@code destFolderNode} including {@code destFolderNode}
80       * and reset the hippotranslation:id property to a newly generate UUID.
81       */
82      protected void resetHippoDocumentTranslationIds() {
83          String destFolderNodePath = null;
84  
85          try {
86              destFolderNodePath = getDestFolderNode().getPath();
87  
88              final Map<String, String> uuidMappings = new HashMap<String, String>();
89  
90              resetHippoTranslatedNodeWithNewUuid(getDestFolderNode(), uuidMappings);
91  
92              JcrTraverseUtils.traverseNodes(getDestFolderNode(),
93                      new NodeTraverser() {
94                          @Override
95                          public boolean isAcceptable(Node node) throws RepositoryException {
96                              return node.isNodeType("hippotranslation:translated");
97                          }
98  
99                          @Override
100                         public boolean isTraversable(Node node) throws RepositoryException {
101                             return !node.isNodeType("hippostdpubwf:document");
102                         }
103 
104                         @Override
105                         public void accept(Node translatedNode) throws RepositoryException {
106                             resetHippoTranslatedNodeWithNewUuid(translatedNode, uuidMappings);
107                         }
108                     });
109         } catch (RepositoryException e) {
110             getLogger().error("Failed to reset hippotranslation:id properties under {}.", destFolderNodePath, e);
111         }
112     }
113 
114     private void resetHippoTranslatedNodeWithNewUuid(final Node translatedNode, final Map<String, String> uuidMappings)
115             throws RepositoryException {
116         if (translatedNode != null && translatedNode.isNodeType("hippotranslation:translated")) {
117             String translationUuid = JcrUtils.getStringProperty(translatedNode, "hippotranslation:id", null);
118 
119             if (UUIDUtils.isValidPattern(translationUuid)) {
120                 String newTranslationUuid;
121 
122                 if (uuidMappings.containsKey(translationUuid)) {
123                     newTranslationUuid = uuidMappings.get(translationUuid);
124                 } else {
125                     newTranslationUuid = UUID.randomUUID().toString();
126                     uuidMappings.put(translationUuid, newTranslationUuid);
127                 }
128 
129                 translatedNode.setProperty("hippotranslation:id", newTranslationUuid);
130             }
131         }
132     }
133 
134     /**
135      * Search all the link holder nodes having hippo:docbase property under destFolderNode
136      * and reset the hippo:docbase properties to the copied nodes under destFolderNode
137      * by comparing the relative paths with the corresponding nodes under the sourceFolderNode.
138      */
139     protected void resetHippoDocBaseLinks() {
140         String destFolderNodePath = null;
141 
142         try {
143             destFolderNodePath = getDestFolderNode().getPath();
144 
145             JcrTraverseUtils.traverseNodes(getDestFolderNode(),
146                     new NodeTraverser() {
147                         private final String sourceFolderBase = getSourceFolderNode().getPath() + "/";
148 
149                         @Override
150                         public boolean isAcceptable(Node node) throws RepositoryException {
151                             return node.isNodeType("hippo:mirror") && node.hasProperty("hippo:docbase");
152                         }
153 
154                         @Override
155                         public boolean isTraversable(Node node) throws RepositoryException {
156                             return !node.isNodeType("hippo:mirror");
157                         }
158 
159                         @Override
160                         public void accept(Node destLinkHolderNode) throws RepositoryException {
161                             String destLinkDocBase = JcrUtils.getStringProperty(destLinkHolderNode, "hippo:docbase", null);
162 
163                             if (StringUtils.isNotBlank(destLinkDocBase)) {
164                                 try {
165                                     Node sourceLinkedNode = getSession().getNodeByIdentifier(destLinkDocBase);
166 
167                                     if (StringUtils.startsWith(sourceLinkedNode.getPath(), sourceFolderBase)) {
168                                         String sourceLinkedNodeRelPath = StringUtils.removeStart(sourceLinkedNode.getPath(),
169                                                 sourceFolderBase);
170                                         Node destLinkedNode = JcrUtils.getNodeIfExists(getDestFolderNode(), sourceLinkedNodeRelPath);
171 
172                                         if (destLinkedNode != null) {
173                                             getLogger().info("Updating the linked node at '{}'.", destLinkHolderNode.getPath());
174                                             destLinkHolderNode.setProperty("hippo:docbase", destLinkedNode.getIdentifier());
175                                         }
176                                     }
177                                 } catch (ItemNotFoundException ignore) {
178                                 }
179                             }
180                         }
181                     });
182         } catch (RepositoryException e) {
183             getLogger().error("Failed to reset link Nodes under destination folder: {}.", destFolderNodePath, e);
184         }
185     }
186 
187     /**
188      * Takes offline all the hippo documents under the {@code destFolderNode}.
189      */
190     protected void takeOfflineHippoDocs() {
191         String destFolderNodePath = null;
192 
193         try {
194             destFolderNodePath = getDestFolderNode().getPath();
195 
196             JcrTraverseUtils.traverseNodes(getDestFolderNode(),
197                     new NodeTraverser() {
198                         @Override
199                         public boolean isAcceptable(Node node) throws RepositoryException {
200                             if (!node.isNodeType("hippostdpubwf:document")) {
201                                 return false;
202                             }
203 
204                             return isLiveVariantNode(node) &&
205                                     StringUtils.equals("published", JcrUtils.getStringProperty(node, "hippostd:state", null));
206                         }
207 
208                         @Override
209                         public boolean isTraversable(Node node) throws RepositoryException {
210                             return !node.isNodeType("hippostdpubwf:document");
211                         }
212 
213                         private boolean isLiveVariantNode(final Node variantNode) throws RepositoryException {
214                             if (variantNode.hasProperty("hippo:availability")) {
215                                 for (Value value : variantNode.getProperty("hippo:availability").getValues()) {
216                                     if (StringUtils.equals("live", value.getString())) {
217                                         return true;
218                                     }
219                                 }
220                             }
221 
222                             return false;
223                         }
224 
225                         @Override
226                         public void accept(Node liveVariant) throws RepositoryException {
227                             liveVariant.setProperty("hippo:availability", ArrayUtils.EMPTY_STRING_ARRAY);
228                             liveVariant.setProperty("hippostd:stateSummary", "new");
229                         }
230                     });
231         } catch (RepositoryException e) {
232             getLogger().error("Failed to take offline link hippostd:publishableSummary nodes under {}.", destFolderNodePath, e);
233         }
234     }
